--Select All
Create Procedure allUsers
As
Select *
From User;
Go

--Select One
Create Procedure singleUser
@User_Id Integer
As
Select *
From User
Where User_ID = @User_id;
Go

--Insert
Create Procedure insertUser
@FirstName Varchar(25),
@LastName Varchar(25),
@Username Varchar(50),
@Password Varchar(50)
As
Insert Into User Values(Next Value For User_ID, FirstName = @FirstName, LastName = @LastName, Username = @Username, Password = @Password);
Go

--Update
Create Procedure updateUser
@User_ID Integer,
@Username Varchar(50),
@Password Varchar(50)
As
Update User
Set Username = FirstName = @FirstName, LastName = @LastName, Username = @Username, Password = @Password
Where User_ID = @User_ID;
Go

--Delete
Create Procedure deleteUser
@User_ID Integer
As
Delete From User
Where User_ID = @User_ID;
Go
