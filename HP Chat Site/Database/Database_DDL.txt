Create Table User
{
    User_ID Integer,
    FirstName Varchar(25),
    LastName Varchar(25),
    Username Varchar(50),
    Password Varchar(50),
    Primary Key (User_ID)
};

Create Table Response
{
    Response_ID Integer,
    Description Varchar(500),
    User_ID Integer,
    Primary Key (Response_ID),
    Foreign Key (User_ID)
};

Create Table Tag
{
    Tag_ID Integer,
    Name Varchar(25),
    Primary Key (Tag_ID)
};

Create Table Response_Tag
{
    Response_ID Integer,
    Tag_ID Integer,
    Primary Key (Response_ID, Tag_ID),
    Foreign Key (Response_ID) References Response,
    Foreign Key (Tag_ID) References Tag
};

Create Sequence User_ID
    Increment By 1
    Start With 1
    NoMaxValue
;

Create Sequence Response_ID
    Increment By 1
    Start With 1
    NoMaxValue
;

Create Sequence Tag_ID
    Increment By 1
    Start With 1
    NoMaxValue
;

--Select All Users
Create Procedure allUsers
As
Select *
From User;
Go

--Select One User
Create Procedure singleUser
@User_Id Integer
As
Select *
From User
Where User_ID = @User_id;
Go

--Insert User
Create Procedure insertUser
@FirstName Varchar(25),
@LastName Varchar(25),
@Username Varchar(50),
@Password Varchar(50)
As
Insert Into User Values(Next Value For User_ID, FirstName = @FirstName, LastName = @LastName, Username = @Username, Password = @Password);
Go

--Update User
Create Procedure updateUser
@User_ID Integer,
@Username Varchar(50),
@Password Varchar(50)
As
Update User
Set Username = FirstName = @FirstName, LastName = @LastName, Username = @Username, Password = @Password
Where User_ID = @User_ID;
Go

--Delete User
Create Procedure deleteUser
@User_ID Integer
As
Delete From User
Where User_ID = @User_ID;
Go
